함수

함수는 프로그램에서 반복적으로 사용되는 기능을 만들어 내기 위한 코드들의 집합입니다.

선언적 함수

선언적 함수는 함수를 선언할 때 이름을 붙여주는 함수를 의미합니다.

function func(){
    실행문;
}

function compute(){
    console.log('선언적 함수');
}

익명 함수

익명 함수는 변수에 데이터를 저장하여 변수를 함수처럼 사용할 수 있게 합니다. 익명 함수는 변수 선언 이후에 호출해야 합니다.

var compute = function(){
    실행문;
};

var compute = function(){
   console.log('익명 함수');
};

즉시 실행 함수

일회성으로 사용되는 함수입니다. 선언과 동시에 함수가 실행되고 함수명이 없기 때문에 재호출이 불가합니다.

(function(){
    실행문;
})();

(function(){
    var sum = 10;
    console.log(sum);
})();

return 함수

함수를 통해 처리된 결과를 반환시켜주는 명령어입니다.

function func(){
       const str = "함수가 실행되었습니다.4";
       return str;
}

함수를 호출할 때 전달하는 변수를 '매개변수'라 합니다.

func("함수가 실행되었습니다.3");

화살표 함수 : 선언적 함수

함수 표현식보다 단순하고 간결한 문법으로 함수를 표현할 수 있는 방법을 말합니다. 화살표 =>를 사용하여 함수를 표현합니다.

//선언적 함수
function func(){
       document.write("함수가 실행되었습니다.1");
}
func();

//화살표 함수1
func = () => {
       document.write("함수가 실행되었습니다.5");
}

//화살표 함수2
       func = () => document.write("함수가 실행되었습니다.5");
func();

화살표 함수 : 익명 함수

// 화살표 함수1
const func = () => {
       document.write("함수가 실행되었습니다.6");
}

// 화살표 함수2
const func = () => document.write("함수가 실행되었습니다.6");

func();

화살표 함수 : 매개변수 함수

// 매개변수 함수
function func(str){
       document.write(str);
}
func("함수가 실행되었습니다.7");

// 화살표 함수1
func = (str) => {
       document.write(str);
}

// 화살표 함수2
func = (str) => document.write(str);

//화살표 함수3
func = str => document.write(str);

func("함수가 실행되었습니다.7");
