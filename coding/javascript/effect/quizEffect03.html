//forEach를 사용하여 버튼(quizConfirm)이 4개 있는것을 알려준다.
quizConfirm.forEach((btn,num) => {
    btn.addEventListener("click", () =>{
    
    //5. 사용자 정답
    const userWord = quizInput[num].value.toLowerCase().trim();

    //6. 사용자 정답과 문제 정답이 같은지 확인한다.
    if(userWord == quizInfo[num].answerResult){
        //정답
        // alert("정답")
        quizView[num].classList.add("like");
        quizConfirm[num].style.display = "none"; //정답 확인하기 버튼 없앰
    } else {
        //오답
        // alert("오답")
        quizView[num].classList.add("dislike");
        quizConfirm[num].style.display = "none"; //정답 확인하기 버튼 없앰
        quizResult[num].style.display = "block"; //정답 해설
        quizInput[num].style.display = "none"; //정답입력칸 삭제

    }
    })
});
